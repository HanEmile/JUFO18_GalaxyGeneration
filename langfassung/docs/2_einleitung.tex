\subsection{Motivation}

\paragraph{ \( \Phi \) }

\begin{equation}
  \Phi(r) = - \frac{4\pi G \rho_0 R_s^3}{r} \ln ( 1+ \frac{r}{R_s} )
\end{equation}

with the limits

\begin{equation}
  \lim_{r\to \infty} \Phi=0
\end{equation}

and

\begin{equation}
  \lim_{r\to 0} \Phi=-4\pi G\rho_0 R_s^2
\end{equation}

\paragraph{ \( \rho \) }

\begin{equation}
  \rho(r) = \frac{1}{\sqrt{2 \cdot \pi} \cdot \sigma} \cdot
  e^{\left( - \frac{(\Phi(r)}{\sigma^{2}} \right)}
\end{equation}
